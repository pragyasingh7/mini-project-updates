data_gt = FileAttachment("diffusion/baseline_benchmarking/gt_metrics.csv").csv({typed: true})
data_gen_full = FileAttachment("diffusion/baseline_benchmarking/full_train_gen_metrics.csv").csv({typed: true})
data_gen_subset = FileAttachment("diffusion/baseline_benchmarking/subset_gen_metrics.csv").csv({typed: true})
combined_bars = FileAttachment("diffusion/baseline_benchmarking/combined_coordination_rings_metrics.csv").csv({typed: true})

filteredData_gt = data_gt.filter(d =>
    Math.abs(d.density - density) < 0.05 &&
    Math.abs(d.temperature - temperature) < 50
)

filteredData_gen_full = data_gen_full.filter(d =>
    Math.abs(d.density - density) < 0.05 &&
    Math.abs(d.temperature - temperature) < 50
)

filteredData_gen_subset = data_gen_subset.filter(d =>
    Math.abs(d.density - density) < 0.05 &&
    Math.abs(d.temperature - temperature) < 50
)

filtered_combined = combined_bars.filter(d =>
    Math.abs(d.density - density) < 0.05 &&
    Math.abs(d.temperature - temperature) < 50
)
Plot.plot({
    title: "Radial Distribution Function (RDF)",
    color: {legend: true, domain: ['GT', 'Gen Full', 'Gen Subset'], range: ['steelblue', 'orange', 'forestgreen']},
    marks: [
        Plot.line(filteredData_gt.filter(d => d.type === "rdf"), {x: "x", y: "y", stroke: (d) => 'GT', tip: true}),
        Plot.line(filteredData_gen_full.filter(d => d.type === "rdf"), {x: "x", y: "y", stroke: (d) => 'Gen Full', tip: true}),
        Plot.line(filteredData_gen_subset.filter(d => d.type === "rdf"), {x: "x", y: "y", stroke: (d) => 'Gen Subset', tip: true}),
        Plot.ruleY([0])
    ]
})
Plot.plot({
  title: "Angular Distribution Function (ADF)",
  color: {legend: true, domain: ['GT', 'Gen Full', 'Gen Subset'], range: ['steelblue', 'orange', 'forestgreen']},
  marks: [
    Plot.line(filteredData_gt.filter(d => d.type === "adf"), {x: "x", y: "y", stroke: (d) => 'GT', tip: true}),
    Plot.line(filteredData_gen_full.filter(d => d.type === "adf"), {x: "x", y: "y", stroke: (d) => 'Gen Full', tip: true}),
    Plot.line(filteredData_gen_subset.filter(d => d.type === "adf"), {x: "x", y: "y", stroke: (d) => 'Gen Subset', tip: true}),
    Plot.ruleY([0])
  ]
})
Plot.plot({
  title: "Coordination Number",
  marks: [
    Plot.barY(filtered_combined.filter(d => d.type === "coordination"), {fx: "x", x: "source", y: "y", fill: "source", tip: true}),
    Plot.ruleY([0])
  ],
  color: {legend: true}
})
Plot.plot({
  title: "Ring Size Distribution",
  marks: [
    Plot.barY(filtered_combined.filter(d => d.type === "rings"), {fx: "x", x: "source", y: "y", fill: "source", tip: true}),
    Plot.ruleY([0])
  ]
})
viewof density = Inputs.range([1.0, 3.5], {step:0.1, label: "Density ((g/cm³))", value: 1.0})
viewof temperature = Inputs.range([2000, 4000], {step: 100, label: "Temperature (K)", value: 2000})
